powershell -NoProfile -ExecutionPolicy Bypass -Command "$ErrorActionPreference='Stop'; $csvPath='D:\PinterestRepo\link\Media_Links.csv'; $repoPath='D:\PinterestRepo'; $destFolder='D:\PinterestRepo\需要补封面的视频'; if (-not (Test-Path $destFolder)) { New-Item -ItemType Directory -Path $destFolder | Out-Null }; $data = Import-Csv -Path $csvPath -Encoding UTF8; $missing = $data | Where-Object { $_.'Media URL' -ne '' -and $_.'Thumbnail' -eq '' }; $count = 0; foreach ($row in $missing) { $vidFile = Get-ChildItem -Path $repoPath -File | Where-Object { $_.BaseName -eq $row.Title -and $_.Extension -match '(?i)\.(mp4|mov|avi|mkv|webm)$' } | Select-Object -First 1; if ($vidFile) { Copy-Item -Path $vidFile.FullName -Destination $destFolder -Force; $count++ } }; Write-Host ('✅ 成功！共捞出 ' + $count + ' 个需要封面的视频，请去文件夹查看：' + $destFolder) -ForegroundColor Green"
